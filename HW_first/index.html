Basic

git config --global user.name blabla- задати ім'я користувача

git config --global user.email blabal@bla- задає пошту користувача

git init - ініціалізація пустого репозиторію;

git status - показує стан репозиторію як файли відслідковуються, які зміни;

git add- додає зміни до файлу або нові файли

git commit- комітить всі зміни і зберігає їх в історію версій ;

git log - показує історію комітів;

git checkout - повертає файли і репозиторій до відповідної версії;

git checkout - повернутись до останньої версії;

git checkout - переключається на вибрану гілку і оновлює папку до її зміненого стану;

git branch - створює нову гілку;

git push -u- відправляє всі закомічені дані на віддалений репозиторій;

git pull - завантажує всі зміни з віддаленого репозиторію і об'єднує їх з локальним

git merge - вносить зміни з віддаленого реппозиторію у поточну гілку локального;

git clone - клонування репозиторію з github;

.gitignore файл

.gitignore - в цей файл вписуємо всі файли та папки за якими не має слідкувати git та не завантажувати у віддалений репозиторій;

Advanced

git revert- відміняє останній коміт, створюючи новий який включає всі відмінені зміни;

git reset - відміняє всі коміти після заданого і залишає всі зміни в робочій папці.;

git rebase - перенести зміни з однієї вітки в іншу;

git cherry-pick- для перенесення окремих комітів в інше місце репозиторію;

git stash- бере ваші зміни, готові і не готові до фіксації, зберігає їх для подальшого використання і забирає їх з робочої області;
